function notifyOwner() { const currentTime = Date.now(); if (currentTime - lastNotifyTime < 60 * 1000) { Swal.fire({ icon: 'warning', title: '发送过于频繁', text: '请稍后再试！', position: 'top', toast: true, timer: 2000, showConfirmButton: false }); return; } // 调用云函数的 URL fetch("https://nuochce-dwmfnmmkvs.cn-hangzhou.fcapp.run", { // 替换为实际的云函数地址 method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ message: "您好，有人需要您挪车，请及时处理。" }) }) .then(response => response.json()) .then(data => { if (data.success) { Swal.fire({ icon: 'success', title: '通知已发送！', text: '车主已收到您的通知。', position: 'top', toast: true, timer: 2000, showConfirmButton: false }); lastNotifyTime = currentTime; } else { Swal.fire({ icon: 'error', title: '通知发送失败', text: data.error || '请稍后重试。', position: 'top', toast: true, timer: 2000, showConfirmButton: false }); } }) .catch(error => { console.error("Error sending notification:", error); Swal.fire({ icon: 'error', title: '网络错误', text: '通知发送出错，请检查网络连接。', position: 'top', toast: true, timer: 2000, showConfirmButton: false }); }); }
